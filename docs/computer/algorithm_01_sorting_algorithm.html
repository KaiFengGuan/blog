十大排序算法总结

GKFTech 2022/3/17 算法JavaScript

十大排序算法（JS简单实现）

# 1 选择排序

排序过程：

  • 首先，找到数组中最小的那个元素；
  • 其次，将它和数组的第 1 个元素交换位置（如果第一个元素就是最小元素那么它就和自己交换）
  • 然后，在剩下的元素中找到最小的元素，将它与数组的第 2 个元素交换位置。
  • 如此往复，直到将整个数组排序。

性质：

  • 时间复杂度：$ O(n^2) $
  • 空间复杂度：$O(1)$
  • 非稳定排序
  • 原地排序
function selectSort(arr) {
    const L = arr.length;
    for (let i = 0; i < L; i++) {
        let idx = i;
        for (let j = i + 1; j < L; j++) {
            if (arr[idx] > arr[j]) idx = j;
        }
        // 交换
        let t = arr[i];
        arr[i] = arr[idx];
        arr[idx] = t;
    }
    return arr;
}

# 2 插入排序

排序过程：

  • 从数组第2个元素开始抽取元素
  • 把它与左边第一个元素比较，如果左边第一个元素比它大，则继续与左边第二个元素比较下去，直到遇到不比它大的元素，然后插到这个元素的右边
  • 继续选取第3，4，….n个元素,重复步骤 2
  • 选择适当的位置插入

性质：

  • 时间复杂度：$ O(n^2) $
  • 空间复杂度：$O(1)$
  • 稳定排序
  • 原地排序
function insertSort(arr) {
    const L = arr.length;
    for (let i = 1; i < L; i++) {
        let inserted = arr[i];
        let j = i - 1;
        while(j >= 0 && arr[j] > inserted) {
            arr[j + 1] = arr[j];
            j--;
        };
        arr[j + 1] = inserted;
    }
    return arr;
}

# 3 冒泡排序

排序过程：

  • 把第 1 个元素与第 2 个元素比较，如果第 1 个比第 2 个大，则交换他们的位置
  • 接着继续比较第 2 个与第 3 个元素，如果第 2 个比第 3 个大，则交换他们的位置….
  • 对每一对相邻元素作同样的工作，从开始第一对到结尾的最后一对，这样一趟比较交换下来之后，排在最右的元素就会是最大的数
  • 除去最右的元素，我们对剩余的元素做同样的工作，如此重复下去，直到排序完成。

性质：

  • 时间复杂度：$ O(n^2) $
  • 空间复杂度：$O(1)$
  • 稳定排序
  • 原地排序
function bubbleSort(arr) {
    const L = arr.length;
    for (let i = 0; i < L; i++) {
        let flag = true; // 优化
        let rightBoundary = L - i - 1;
        for (let j = 0; j < rightBoundary; j++) {
            if (arr[j] > arr[j + 1]) {
                flag = false;
                [arr[j + 1], arr[j]] = [arr[j], arr[j + 1]];
            }
        }
        if (flag) break;
    }
    return arr;
}

# 4 希尔排序

希尔排序可以说是插入排序的一种变种。

  • 简单插入排序很循规蹈矩，不管数组分布是怎么样的，依然一步一步的对元素进行比较，移动，插入，比如[5,4,3,2,1,0]这种倒序序列，数组末端的0要回到首位置很是费劲，比较和移动元素均需n-1次。

希尔排序是把记录按下标的一定增量分组，对每组使用直接插入排序算法排序；随着增量逐渐减少，每组包含的关键词越来越多，当增量减至1时，整个文件恰被分成一组，算法便终止。

排序过程：

  • 选择增量gap=length/2，缩小增量继续以gap = gap/2的方式，这种增量选择我们可以用一个序列来表示，{n/2,(n/2)/2...1}，称为增量序列。
  • 希尔排序的增量序列的选择与证明是个数学难题，我们选择的这个增量序列是比较常用的，也是希尔建议的增量，称为希尔增量，但其实这个增量序列不是最优的。此处我们做示例使用希尔增量。

性质：

  • 时间复杂度：$ O(nlog(n)) $
  • 空间复杂度：$O(1)$
  • 非稳定排序
  • 原地排序
function shellSort(arr) {
    let L = arr.length;
    for (let gap = Math.floor(L / 2); gap > 0; gap = Math.floor(gap / 2)) {
        for (let i = gap; i < L; i++) {
            insertI(arr, gap, i);	// 将arr[i]插入到所在分组正确的位置
        }
    }
    return arr;
}

function insertI(arr, gap, i) {
    let inserted = arr[i];
    let j = i - gap;
    while(j >= 0 && arr[j] > inserted) {
        arr[j + gap] = arr[j];
        j -= gap;
    }
    arr[j + gap] = inserted;
}

# 5 归并排序

排序过程：

  • 将一个大的无序数组有序，我们可以把大的数组分成两个，然后对这两个数组分别进行排序，之后在把这两个数组合并成一个有序的数组。由于两个小的数组都是有序的，所以在合并的时候是很快的。
  • 通过递归的方式将大的数组一直分割，直到数组的大小为 1，此时只有一个元素，那么该数组就是有序的了，之后再把两个数组大小为1的合并成一个大小为2的，再把两个大小为2的合并成4的 ….. 直到全部小的数组合并起来。

性质：

  • 时间复杂度：$ O(nlog(n)) $
  • 空间复杂度：$O(n)$
  • 稳定排序
  • 非原地排序
function mergeSort(arr, left, right) {
    if (left >= right) return;
    
    let mid = Math.floor((left + right) / 2);
    mergeSort(arr, left, mid);
    mergeSort(arr, mid + 1, right);
    merge(arr, left, mid, right);
}
function merge(arr, left, mid, right) {
    let i = left, j = mid + 1;
    let temp = arr.slice(left, right + 1);
    for (let k = left; k <= right; k++) {
        if (i > mid) 		temp[k] = arr[j++];
        else if (j > right) temp[k] = arr[i++];
        else if (arr[i] < arr[j]) temp[k] = arr[i++];
        else	temp[k] = arr[j++];
    }
    for (let k = left; k <= right; k++) {
        arr[k] = temp[k];
    }
}

function mergeSort(arr) {
    const L = arr.length;
    // i 是子数组大小。 1，2，4，8......
    for (let i = 1; i < L; i += i) {
        let left = 0;
        let mid = left + i - 1;
        let right = mid + i;
        while (right < L) {
            merge(arr, left, mid, right);
            left = right + 1;
            mid = left + i - 1;
            right = mid + i;
        }
        if (left < L && mid < L) {
            merge(arr, left, mid, L - 1);
        }
    }
    return arr;
}

# 6 快速排序

选择一个基准数，通过一趟排序将要排序的数据分割成独立的两部分：其中一部分的所有数据都比另外一部分的所有数据都要小。然后，再按此方法对这两部分数据分别进行快速排序，整个排序过程可以递归进行，以此达到整个数据变成有序序列。

快速排序流程：

  • 从数列中挑出一个基准值；
  • 将所有比基准值小的摆放在基准前面，所有比基准值大的摆在基准的后面（相同的数可以到任一边）；
  • 在这个分区退出之后，该基准就处于数列的中间位置。
  • 递归地把“基准值前面的子数列"和"基准值后面的子数列”进行排序。

性质：

  • 时间复杂度：$ O(nlog(n)) $，最坏$O(n^2)$
  • 空间复杂度：$O(log(n))$
  • 非稳定排序
  • 原地排序
function partition(arr, left, right) {
    let temp = arr[left];
    let i = left + 1, j = right;
    while (true) {
        while (i <= j && arr[i] <= temp) i++;	// 向 右 找第一个 大于 基准值的元素
        while (i <= j && arr[j] >= temp) j--;	// 向 左 找第一个 小于 基准值的元素
        if (i >= j) break;
        [arr[i], arr[j]] = [arr[j], arr[i]];
    }
    arr[left] = arr[j];
    arr[j] = temp;
    return j;
}

function quickSort(arr, left, right) {
    if (left >= right) return;

    let center = partition(arr, left, right);
    quickSort(arr, left, center - 1);	// 左边部分
    quickSort(arr, center + 1, right);	// 右边部分
}

# 7 堆排序

堆是一个完全二叉树。堆由数组来实现，相当于对完全二叉树做层序遍历。

堆排序过程：

  • 将初始二叉树转化为大顶堆（heapify）（实质是从最后一个叶子节点的父节点开始，从下至上，从右至左，对每一个非叶子结点做heapify操作），此时根结点为最大值
  • 将根节点与最后一个结点交换
  • 除开最后一个结点，将其余节点组成的新堆转化为大顶堆（heapify），此时根结点为次最大值
  • 重复步骤2-3，直到堆中元素个数为1（或其对应数组的长度为1），排序完成。

对于用数组表示的完全二叉树，第 i 个节点，其：

  • parent：$(i - 1) / 2$
  • child1：$2*i + 1$
  • child2：$2*i + 2$

性质：

  • 时间复杂度：$ O(nlog(n)) $
  • 空间复杂度：$O(1)$
  • 非稳定排序
  • 原地排序
// 对以 i 节点为父节点的 tree 进行 heapfiy 操作
function heapfiy(tree, N, i) {
  let c1 = 2 * i + 1;
  let c2 = 2 * i + 2;
  let max = i;
  if (c1 < N && tree[c1] > tree[max]) {
    max = c1;
  }
  if (c2 < N && tree[c2] > tree[max]) {
    max = c2;
  }
  if (max !== i) {
    [tree[i], tree[max]] = [tree[max], tree[i]];
    heapfiy(tree, N, max);
  }
}

function heapSort(arr) {
  // heapfiy(arr, arr.length, 0);
  let N = arr.length;

  // 构建 heap
  let i = Math.floor((N - 1) / 2); // 求最后一个节点的父节点, 即最后一个非叶子节点
  for (; i >= 0; i--) {
    heapfiy(arr, N, i);
  }

  // 排序
  for (let i = N - 1; i >= 0; i--) {
    [arr[0], arr[i]] = [arr[i], arr[0]];
    heapfiy(arr, i, 0);
  }
}

# 8 计数排序

计数排序是一种适合于最大值和最小值的差值不是很大的排序。

排序过程：

  • 就是把数组元素作为数组的下标，然后用一个临时数组统计该元素出现的次数，例如 temp[i] = m, 表示元素 i 一共出现了 m 次。
  • 最后再把临时数组统计的数据从小到大汇总起来，此时汇总起来是数据是有序的。

性质（注：K表示临时数组的大小）：

  • 时间复杂度：$ O(n+k) $
  • 空间复杂度：$O(k)$
  • 稳定排序
  • 非原地排序
function countSort(arr) {
  let N = arr.length;

  let max = arr[0];
  let min = arr[0];
  for (let i = 1; i < N; i++) {
    if (arr[i] > max) max = arr[i];
    if (arr[i] < min) min = arr[i];
  }

  const count = new Array(max - min + 1).fill(0);
  for (let i = 0; i < N; i++) {
    count[arr[i] - min]++;
  }
  
  let k = 0;
  for (let i = 0; i < count.length; i++) {
    while(count[i]-- > 0) {
      arr[k++] = i + min;
    }
  }
}

# 9 桶排序

  • 桶排序就是把最大值和最小值之间的数进行划分，例如分成 10 个区间，10个区间对应10个桶
  • 我们把各元素放到对应区间的桶中去，再对每个桶中的数进行排序
  • 桶中的数可以采用归并排序，也可以采用快速排序之类的
  • 之后每个桶里面的数据就是有序的了，我们再进行合并汇总

性质：

  • 时间复杂度：$ O(n+k) $
  • 空间复杂度：$O(n+k)$
  • 稳定排序
  • 非原地排序
function bucketSort(arr) {
  let N = arr.length;
  let num = 3;  // 桶的个数

  let max = arr[0];
  let min = arr[0];
  for (let i = 1; i < N; i++) {
    if (arr[i] > max) max = arr[i];
    if (arr[i] < min) min = arr[i];
  }
  let d = max - min;
  let range = Math.ceil(d / num) || 1; // 每个桶存放的数值范围，至少为1
  // 创建二维数组，第一维表示第几个桶，第二维表示该桶里存放的数
  let bucket = Array.from(Array(num)).map(() => Array().fill(0));

  for (let i = 0; i < N; i++) {
    let index = parseInt((arr[i] - min) / range); // 计算元素应该分布在哪个桶
    index = index >= num ? num - 1 : index; // 防止index越界

    let temp = bucket[index];
    let j = temp.length - 1;
    while(j >= 0 && arr[i] < temp[j]) {
      temp[j + 1] = temp[j];
      j--;
    }
    temp[j+1] = arr[i];
  }

  let k = 0;
  for (let i = 0; i < num; i++) {
    for (let j = 0; j < bucket[i].length; j++) {
      arr[k++] = bucket[i][j];
    }
  }
}

# 10 基数排序

基数排序的排序思路是这样的：

  • 先以个位数的大小来对数据进行排序，接着以十位数的大小来多数进行排序，接着以百位数的大小……
  • 排到最后，就是一组有序的元素了。不过，他在以某位数进行排序的时候，是用“桶”来排序的。
  • 由于某位数（个位/十位….，不是一整个数）的大小范围为0-9，所以我们需要10个桶，然后把具有相同数值的数放进同一个桶里，之后再把桶里的数按照0号桶到9号桶的顺序取出来，这样一趟下来，按照某位数的排序就完成了

性质：

  • 时间复杂度：$ O(kn) $
  • 空间复杂度：$O(n+k)$
  • 稳定排序
  • 非原地排序
function radixSort(arr) {
  let N = arr.length;

  let max = arr[0];
  for (let i = 1; i < N; i++) {
    if (arr[i] > max) max = arr[i];
  }
  let num = max.toString().length;
  let bucket = Array.from(Array(10)).map(() => Array().fill(0));// 创建10个桶

  for (let i = 1; i <= num; i++) {
    let base = Math.pow(10, i - 1);
    for (let j = 0; j < N; j++) {
      let index = parseInt(arr[j] / base) % 10;
      bucket[index].push(arr[j]);
    }

    let k = 0;
    for (let j = 0; j < bucket.length; j++) {
      while (bucket[j].length) {
        arr[k++] = bucket[j].shift();
      }
    }
  }
}
