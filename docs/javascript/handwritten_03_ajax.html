如何封装一个Ajax

GKFTech 2022/4/5 JavaScript手撕

Ajax 的全称是Asynchronous JavaScript and XML（异步的JavaScript 和 XML），其中，Asynchronous 是 异步 的意思，它有别于传统web开发中采用的同步的方式。

Ajax包含的技术：

  1. 使用CSS和XHTML来表示。
  2. 使用DOM模型来交互和动态显示。
  3. 使用XMLHttpRequest来和服务器进行异步通信。(最重要)
  4. 使用javascript来绑定和调用。

# AJAX原理

Ajax的原理：简单来说通过 XMLHttpRequest 对象来向服务器发送异步请求，从服务器获得数据，然后用javascript来操作DOM而更新页面。最关键的一步就是从服务器获得请求数据。

XMLHttpRequest 对象的属性和方法学习 看这里 (opens new window)

# AJAX使用步骤

  • 第一步：创建XMLHttpRequest对象
  • 第二步：创建和指定响应函数
  • 第三步：打开连接（指定URL和请求方法）
  • 第四步：发送请求

第三步是使用XMLHttpRequest对象的open()方法，字面意思open是打开的意思，即打开连接。但是准确的说应该是指定Http请求。因为浏览器在使用AJAX技术与服务通信时，发送的是Http请求，那么就要指定Http的请求方法，url等信息。

# 实现

# ES5

// ES5 
function ajax(options) {
  options = Object.assign({
    url: '',
    method: 'get',
    data: null,
		success: function(res) { return res },
    error: function(err) { return err },
  }, options);

  // 1. 创建一个异步对象
  var xmlhttp = null;
  if (window.XMLHttpRequest) {
    xmlhttp = new XMLHttpRequest();
  } else if (window.ActiveXObject) {
    xmlhttp = new ActiveXObject('Microsoft.XMLHTTP');
  }

  if (!xmlhttp) {
    console.log('Your browser does not support XMLHTTP');
    return;
  }

  // 2. 创建和指定响应函数
  xmlhttp.onreadystatechange = function () {
    if (xmlhttp.readyState === 4) {
      if (xmlhttp.status >= 200 && xmlhttp.status < 300 || xmlhttp.status === 304) {
        console.log('返回数据了');
        options.success(xmlhttp.responseText);
      } else {
        console.log('失败了')
        options.error(new Error(xmlhttp.responseText));
      }
    }
  };

  // 3. 打开链接
  // 4. 发送请求
  var type = options.method.toLowerCase();
  if (type === 'get') {
    xmlhttp.open(type, options.url, true);
    xmlhttp.send(null);
  }
  else {
    xmlhttp.open(type, options.url, true);
    xmlhttp.setRequestHeader('Content-type', 'application/x-www-form-urlencoded');
    xmlhttp.send(JSON.stringify(options.data));
  }
  
  // 超时处理
  var timer;
  if (options.timeout) {
    timer = setInterval(function() {
      xmlhttp.abort();
      clearInterval(timer);
    }, options.timeout);
  }
}

# ES6

function ajax({
  url = '',
  method = 'get',
  data = null,
} = {}) {
  return new Promise((resolve, reject) => {
    // 1. 创建一个异步对象
    const xmlhttp = new XMLHttpRequest();

    // 2. 创建和指定响应函数
    xmlhttp.onreadystatechange = function () {
      if (xmlhttp.readyState === 4) {
        if (xmlhttp.status === 200 || xmlhttp.status === 304) {
          resolve(xmlhttp.responseText);
        } else {
          reject(xmlhttp.responseText);
        }
      }
    };

    // 3. 打开链接
    // 4. 发送请求
    const type = method.toLowerCase();
    if (type === 'get') {
      xmlhttp.open(type, url, true);
      xmlhttp.send(null);
    }
    else {
      xmlhttp.open(type, url, true);
      xmlhttp.setRequestHeader('Content-type', 'application/json');
      xmlhttp.send(JSON.stringify(data));
    }
  });
}
