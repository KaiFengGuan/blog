# 1 防抖&节流

函数防抖和函数节流：优化高频率执行js代码的一种手段。

js中的一些事件如浏览器的resize、scroll，鼠标的mousemove、mouseover，input输入框的keypress等事件在触发时，会不断地调用绑定在事件上的回调函数，极大地浪费资源，降低前端性能。为了优化体验，需要对这类事件进行调用次数的限制。

# 1.1 防抖

  • 在事件被触发n秒后再执行回调，如果在这n秒内又被触发，则重新计时。

  • 为了方便防抖函数的调用和回调函数fn的传参问题，需要使用闭包来优化封装函数。

function debounce(fn, delay) {
    var timer; // 维护一个 timer
    return function () {
        var _this = this; // 取debounce执行作用域的this
        var args = arguments;
        if (timer) {
            clearTimeout(timer);
        }
        timer = setTimeout(function () {
            fn.apply(_this, args); // 用apply指向调用debounce的对象，相当于_this.fn(args);
        }, delay);
    };
}

# 1.2 节流

每隔一段时间，只执行一次函数。

  • 定时器实现节流函数：

    function throttle(fn, delay) {
    var timer;
    return function () {
        var _this = this;
        var args = arguments;
        if (timer) {
            return;
        }
        timer = setTimeout(function () {
            fn.apply(_this, args);
            timer = null; // 在delay后执行完fn之后清空timer，此时timer为假，throttle触发可以进入计时器
        }, delay)
    }
}

# 1.3 比较

  • 相同点：

    • 都可以通过使用 setTimeout 实现。
    • 目的都是，降低回调执行频率。节省计算资源。

  • 不同点：

    • 函数防抖，在一段连续操作结束后，处理回调，利用clearTimeout 和 setTimeout实现。
    • 函数节流，在一段连续操作中，每一段时间只执行一次，频率较高的事件中使用来提高性能。
    • 函数防抖关注一定时间连续触发的事件只在最后执行一次。
    • 函数节流侧重于一段时间内只执行一次。

# 1.4 应用场景

  • 防抖场景：连续的事件，只需触发一次回调的场景有：

    • 搜索框搜索输入。只需用户最后一次输入完，再发送请求
    • 手机号、邮箱验证输入检测
    • 窗口大小Resize。只需窗口调整完成后，计算窗口大小。防止重复渲染。

  • 函数节流：间隔一段时间执行一次回调的场景有：

    • 滚动加载，加载更多或滚到底部监听
    • 谷歌搜索框，搜索联想功能
    • 高频点击提交，表单重复提交

# 2 深浅拷贝

浅拷贝和深拷贝都只针对于引用数据类型：

  • 浅拷贝只复制指向某个对象的指针，而不复制对象本身，新旧对象还是共享同一块内存；
  • 深拷贝会另外创造一个一模一样的对象，新对象跟原对象不共享内存，修改新对象不会改到原对象；
  • 浅拷贝只复制对象的第一层属性
  • 深拷贝可以对对象的属性进行递归复制

# 2.1 浅拷贝

（1）使用 Object.assign()

Object.assign()拷贝的是（可枚举）属性值，假如源值是一个对象的引用，它仅仅会复制其引用值。

let obj1 = { a: 0 , b: { c: 0}};
let obj2 = Object.assign({}, obj1);

obj1.a = 1;
obj1 // { a: 1, b: { c: 0}}
obj2 // { a: 0, b: { c: 0}}

obj2.a = 2;
obj1 // { a: 1, b: { c: 0}}
obj2 // { a: 2, b: { c: 0}}

obj2.b.c = 3;

推荐阅读

（1）递归

function deepClone(obj) {
    if (!obj && typeof obj !== "object") return obj;

    let objClone = Array.isArray(obj) ? [] : {};
    for(let key in obj) {
        if(obj.hasOwnProperty(key)){
            // 判断ojb子元素是否为对象，如果是，递归复制
            if(obj[key] && typeof obj[key] === "object") {
                objClone[key] = deepClone(obj[key]);
            } else {
                // 如果不是，简单复制
                objClone[key] = obj[key];
            }
        }
    }
    return objClone;
}

function deepClone_1(obj, hash = new WeakMap()) {
  if (!isObject(obj)) return obj;

  if (hash.has(obj)) return hash.get(obj);
  const newObj = Array.isArray(obj) ? [] : {};
  hash.set(obj, newObj);
  Reflect.ownKeys(obj).forEach(key => {
    if (isObject(obj[key])) {
      newObj[key] = deepClone_1(obj[key], hash);
    } else {
      newObj[key] = obj[key];
    }
  });
  return newObj;
}

# 3 call、apply、bind

# 3.1 call

  1. 获取第一个参数（注意第一个参数为null或undefined时，this指向window），构建对象
  2. 将对应函数传入该对象中
  3. 获取参数并执行相应函数
  4. 删除该对象中函数，消除副作用
  5. 返回结果
Function.prototype.myCall = function (thisArg, ...params) {
  if (thisArg === null || typeof thisArg === "undefined") return this(params);      // 处理 null || undefined
  if (!/^(object|function)$/i.test(typeof thisArg)) thisArg = new Object(thisArg);  // 原始类型的值 转为包装类

  const sb = Symbol('fn');
  Object.defineProperty(thisArg, sb, {  // thisArg[sb] = this;
    value: this,
    configurable: true,
  });
  const res = thisArg[sb](...params);
  delete thisArg[sb];

  return res;
}

# 3.2 apply

apply和call的区别主要是参数的不同（call是参数列表，apply是参数数组），所以其实现步骤的call大体类似。

Function.prototype.myApply = function (context, args) {
  // 处理 null || undefined
  if (context === null || typeof context === "undefined") context = window;
  
  let fn = Symbol();
  context[fn] = this;

  const res = args ? context[fn](...args) : context[fn]();
  delete context[fn];

  return res;
}

# 3.3 bind

第一种情况：若是将 bind 绑定之后的函数当作构造函数，通过 new 操作符使用，则不绑定传入的 this，而是将 this 指向实例化出来的对象

  • 此时由于new操作符作用，this指向result实例对象，而result又继承自传入的_this
  • 根据原型链知识可得出以下结论：
    • this.__proto__ === result.prototype，this instanceof result为 true
    • this.__proto__.__proto__ === result.prototype.__proto__ === _this.prototype，this instanceof _this为 true
Function.prototype.myBind = function (context, ...args) {
  // 处理 null || undefined
  if (context === null || typeof context === "undefined") context = window;

  let fn = Symbol();
  context[fn] = this;
  let _this = this;

  const result = function (...innerArgs) {
    if (this instanceof _this === true) {
      // 此时this指向指向result的实例  这时候不需要改变this指向
      this[fn] = _this;
      this[fn](...[...args, ...innerArgs]);
    } else {
      // 如果只是作为普通函数调用  那就很简单了 直接改变this指向为传入的context
      context[fn](...[...args, ...innerArgs]);
    }
  };

  // 如果绑定的是构造函数 那么需要继承构造函数原型属性和方法
  // 实现继承的方式: Object.create()方法创建一个新对象，使用现有的对象来提供新创建的对象的__proto__
  //即 result实例化后的__proto__ 指向this.prototype
  result.prototype = Object.create(this.prototype);
  return result;
}

# 4 手写new

先前的对象或原型学习笔记 (opens new window)

  1. 一个新对象被创建；
  2. 该对象的__proto__属性指向该构造函数的原型，即fn.prototype；
  3. 将执行上下文（this）绑定到新创建的对象中；
  4. 如果构造函数有返回值（对象或函数），那么这个返回值将取代第一步中新创建的对象。
function myNew(fn, ...args) {
  const obj = Object.create(fn.prototype);

  const result = fn.apply(obj, args);
  if (result && (typeof result === 'object' || typeof result === 'function')) {
    return result;
  }
  return obj;
}

# 5 手写setInterval

function mySetInterval(fn, delay = 1000) {
  let timer = null;
  let clear = false;
  function interval() {
    if (clear) {
      clear = false;
      clearTimeout(timer);  // 清除计时器
      return;
    }
    fn();
    setTimeout(interval, delay);
  }
  timer = setTimeout(interval, delay);

  return () => {  // 返回一个函数可以取消interval
    clear = true;
  }
}

# 6 手写柯里化

什么是柯里化？

在计算机科学中，柯里化（Currying）是把接受多个参数的函数变换成接受一个单一参数（最初函数的第一个参数）的函数，并且返回接受余下的参数且返回结果的新函数的技术。

函数柯里化的主要作用和特点就是参数复用、提前返回和延迟执行。

function curry(fn, ...args) {
  const n = fn.length;
  args = args || [];
  return function() {
    const newArgs = args.concat(Array.prototype.slice.call(arguments));
    if (newArgs.length < n) {
      return curry.call(this, fn, ...newArgs);
    } else {
      return fn.call(this, ...newArgs);
    }
  }
}

# 7 发布订阅模式

class Event {
  constructor() {
    this.events = {};
  }

  on(type, fn) {
    this.events[type] = this.events[type] || [];
    this.events[type].push(fn);
  }

  emit(type, ...args) {
    const fns = this.events[type];
    if (!fns) return;
    fns.forEach(fn => fn(...args));
  }

  off(type, fn) {
    if (!fn) {
      delete this.events[type];
      return;
    };
    this.events[type] = this.events[type].filter(item => item !== fn);
  }

  once(type, fn) {
    const onceFn = (...args) => {
      this.off(type, onceFn);
      fn(...args)
    }
    this.on(type, onceFn);
  }
}

# 8 对象展开

function isObject(val) {
  return typeof val === 'object' && val !==null;
}

function objFlatten(obj) {
  if (!isObject(obj)) return;

  const res = {};
  const dfs = (val, prefix) => {
    if (isObject(val)) {
      if (Array.isArray(val)) {
        val.forEach((d, i) => {
          dfs(d, `${prefix}[${i}]`);
        });
      } else {
        for (const key in val) {
          dfs(val[key], `${prefix}${prefix ? '.' : ''}${key}`);
        }
      }
    } else {
      res[prefix] = val;
    }
  }

  dfs(obj, '');
  return res;
}

# 9.4 splice

参数说明：Array.prototype.splice() (opens new window)

Array.prototype.mySplice = function(start, deleteCount, ...items) {
  const oldArr = this;
  if (start < 0) {
    start = oldArr.length + start;
    if (start < 0) start = 0;
  }

  let end;  // 指向移除的最后一个元素的后一个
  if (deleteCount <= 0) { // 不移除元素
    end = start;
  } else if (!deleteCount) { // start后全部移除
    end = oldArr.length;
  } else {
    end = start + deleteCount;
    if (end > oldArr.length) end = oldArr.length;
  }

  const beforeArr = [];
  const afterArr = [];
  const returnArr = [];
  for (let i = 0; i < oldArr.length; i++) {
    if (i < start) {
      beforeArr.push(oldArr[i]);
    }
    if (i >= end) {
      afterArr.push(oldArr[i]);
    }
    if (i >= start && i < end) {
      returnArr.push(oldArr[i]);
    }
  }
  oldArr.length = 0;  // 清空原数组
  [...beforeArr, ...items, ...afterArr].forEach((d, i) => oldArr[i] = d);
  return returnArr;
}

Array.prototype.mySlice = function(begin, end) {
  const oldArr = this;
  const length = oldArr.length;
  
  if (!begin) {
    begin = 0;
  } else if (begin < 0) {
    begin = length + begin;
    if (begin < 0) begin = 0;
  }

  if (!end || end > length) {
    end = length;
  } else if (end < 0) {
    end = length + end;
  }

  const res = [];
  for (let i = begin; i < end; i++) {
    res.push(oldArr[i]);
  }
  return res;
}

# 10 compose

compose 是函数式编程中一个非常重要的函数，compose的函数作用就是组合函数的，将函数串联起来执行。将多个函数组合起来，一个函数的输出记过是另一个函数的输入参数，一旦第一个函数开始执行，就会像多米诺骨牌一样推导执行。

  return x + 3;

  return x + 4;

const a = compose(fn1, fn2, fn3, fn4);
console.log(a(1)); // 1+4+3+2+1=11

function compose(...fns) {
  if (fns.length === 0) return (v) => v;
  if (fns.length === 1) return fns[0];
  return fns.reduce(
    (pre, cur) => (...args) => pre(cur(...args))
  );
}

/**
 * @param {Array: Function} funs 元素是一个返回promise的函数
 * @param {Number} max 
 * @returns Promise
 */
function requestLimition(funs, max = 4) {
  return new Promise((resolve, reject) => {
    const len = funs.length;
    const result = [];
    let idx = 0;
    let count = 0;
    const start = () => {
      while (idx < len && max > 0) {
        max--;
        console.log(`${idx} start...`)
        const fn = funs[idx++];
        fn().then(res => {
          max++;
          result[count++] = res;
          if (count === len) resolve(result);
          else start();
        });
      }
    }
    start();
  });
}

# 13 手写Object.is

function ObjectIs(x, y) {
  if (x === y) {
    // x 不等于 0，直接返回true
    // x 等于 0，判断是否是 +0 或 -0
    return x !== 0 || (1 / x === 1 / y);
  }

  // x和y都是NaN的情况
  return x !== x && y !== y;
}

console.log(ObjectIs(NaN, NaN))     // true
console.log(ObjectIs(+0, -0))       // false
console.log(ObjectIs('aaa', 'aaa')) // true

# 14 对象展开

function isObject(val) {
  return typeof val === 'object' && val !==null;
}

function objFlatten(obj) {
  if (!isObject(obj)) return;

  const res = {};
  const dfs = (val, prefix) => {
    if (isObject(val)) {
      if (Array.isArray(val)) {
        val.forEach((d, i) => {
          dfs(d, `${prefix}[${i}]`);
        });
      } else {
        for (const key in val) {
          dfs(val[key], `${prefix}${prefix ? '.' : ''}${key}`);
        }
      }
    } else {
      res[prefix] = val;
    }
  }

  dfs(obj, '');
  return res;
}

# 15 列表转为树结构

将列表结构转换为树结构，parentId为0则是根节点

function ListToTree(list) {
  const obj = {};
  for (let i = 0; i < list.length; i++) {
    obj[list[i].id] = list[i];
  }

  let tree;
  for (let key in obj) {
    const item = obj[key];
    if (item.parentId !== 0) {
      if (!obj[item.parentId].children) {
        obj[item.parentId].children = [];
      }
      obj[item.parentId].children.push(item);
    } else {
      tree = item;
    }
  }
  return tree;
}

# 16 树结构转为列表

function TreeToList(tree) {
  const list = [];

  const travel = (tree) => {
    if (!tree.children) return;

    let children = tree.children;
    children.forEach(item => {
      list.push(item);
      travel(item);
    });
    delete tree.children;
  }

  list.push(tree);
  travel(tree);
  delete tree.children;
  return list;
}
